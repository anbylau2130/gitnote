from tSalSalePay201811 b,Tsalsale201811 a,tSmdCobOrg org
 where a.billno=b.billno
--and a.saleno='20181112012902'
and b.udp1 in (20,80)
AND a. bizdate BETWEEN '2018-11-11' AND '2018-11-18'
AND a.orgcode=org.orgcode
